What are Variables in JavaScript?

Variables are containers for storing data values. In JavaScript, you can think of a variable as a labeled box where you can store information to use later in your code.

Why Use Variables?

  • Reusability: Store values once, use them multiple times
  • Readability: Give meaningful names to values
  • Maintainability: Change value in one place, update everywhere
  • Dynamic Behavior: Store different values at different times
  • Memory Management: Allocate and manage memory for data

Variable Naming Rules

  • Names can contain letters, digits, underscores, and dollar signs
  • Names must begin with a letter, $ or _
  • Names are case sensitive (myVar â‰  myvar)
  • Reserved words cannot be used as names
  • Use camelCase for variable names (e.g., myVariableName)

Variable Declaration: var, let, const

JavaScript provides three ways to declare variables, each with different characteristics regarding scope, hoisting, and reassignment.

Declaration Keywords

// 1. var - The original way (function-scoped)
var oldVariable = "I'm function-scoped";
oldVariable = "I can be reassigned"; // Reassignment allowed

// 2. let - Modern variable declaration (block-scoped)
let modernVariable = "I'm block-scoped";
modernVariable = "I can also be reassigned";

// 3. const - Constant declaration (block-scoped)
const constantVariable = "I cannot be reassigned";
// constantVariable = "New value"; // Error! Cannot reassign const

// Declaration without initialization
let uninitialized; // undefined
uninitialized = "Now I have a value";

// Multiple declarations
let a = 1, b = 2, c = 3;

Comparison Table

Keyword Scope Hoisting Reassignable Redeclarable Use When
var Function scope Yes (undefined) Yes Yes Legacy code, avoid in modern JS
let Block scope Yes (TDZ*) Yes No Variables that change value
const Block scope Yes (TDZ*) No No Values that shouldn't change

* TDZ = Temporal Dead Zone (cannot access before declaration)

Best Practice

Use const by default, let when you need to reassign, and avoid var in modern JavaScript. This makes your code more predictable and easier to debug.

JavaScript Data Types

JavaScript has dynamic types: the same variable can be used to hold different data types. There are 7 primitive data types and Objects.

Primitive vs Reference Types

// PRIMITIVE TYPES (stored by value)
let a = 10;
let b = a;  // b gets a COPY of a's value
a = 20;
console.log(a);  // 20
console.log(b);  // 10 (unchanged!)

// REFERENCE TYPES (stored by reference)
let obj1 = { value: 10 };
let obj2 = obj1;  // obj2 gets a REFERENCE to obj1
obj1.value = 20;
console.log(obj1.value);  // 20
console.log(obj2.value);  // 20 (changed too!)

// Arrays are also reference types
let arr1 = [1, 2, 3];
let arr2 = arr1;
arr1.push(4);
console.log(arr2);  // [1, 2, 3, 4] (changed too!)

Type Checking with typeof

// Using typeof operator
console.log(typeof "Hello");        // "string"
console.log(typeof 42);            // "number"
console.log(typeof 3.14);         // "number"
console.log(typeof true);         // "boolean"
console.log(typeof undefined);    // "undefined"
console.log(typeof null);          // "object" (historical bug!)
console.log(typeof { name: "John" });  // "object"
console.log(typeof [1, 2, 3]);      // "object" (not "array"!)
console.log(typeof function() {}); // "function"

// Better array checking
console.log(Array.isArray([1, 2, 3]));  // true
console.log(Array.isArray({ length: 3 }));  // false

// Checking for null specifically
let value = null;
console.log(value === null);  // true (direct comparison)

The typeof null Bug

In JavaScript, typeof null returns "object". This is a historical bug that remains for backward compatibility. Always use value === null to check for null values.

Scope Examples

// Global scope
const GLOBAL_VAR = "I'm accessible everywhere";

function testFunction() {
  // Function scope (var)
  var functionScoped = "I exist throughout the function";
  
  if (true) {
    // Block scope (let/const)
    let blockScoped = "I exist only in this block";
    var stillFunctionScoped = "I'm function-scoped";
    
    console.log(blockScoped);  // Works
    console.log(GLOBAL_VAR);  // Works
  }
  
  // console.log(blockScoped); // Error! blockScoped is not defined
  console.log(stillFunctionScoped);  // Works
}

testFunction();
// console.log(functionScoped); // Error! functionScoped is not defined
console.log(GLOBAL_VAR);  // Works

Hoisting Examples

// HOISTING: JavaScript moves declarations to the top

// Variable hoisting (var)
console.log(hoistedVar);  // undefined (not an error!)
var hoistedVar = "I'm hoisted";
console.log(hoistedVar);  // "I'm hoisted"

// Variable hoisting (let/const) - Temporal Dead Zone
// console.log(hoistedLet); // Error! Cannot access before initialization
let hoistedLet = "I'm in TDZ before declaration";

// Function hoisting
sayHello();  // Works! Function declarations are fully hoisted

function sayHello() {
  console.log("Hello!");
}

// Function expression hoisting (different!)
// sayGoodbye(); // Error! sayGoodbye is undefined

var sayGoodbye = function() {
  console.log("Goodbye!");
};

sayGoodbye();  // Works

Hoisting Best Practices

  1. Always declare variables at the top of their scope
  2. Use let/const instead of var to avoid unexpected behavior
  3. Initialize variables when you declare them
  4. Use function declarations (not expressions) if you need hoisting

Type Conversion & Coercion

JavaScript can convert values from one type to another, either explicitly (type conversion) or implicitly (type coercion).

Explicit Type Conversion

// String to Number
let str = "123";
let num1 = Number(str);        // 123
let num2 = parseInt(str);     // 123
let num3 = parseFloat("3.14"); // 3.14
let num4 = +str;               // 123 (unary plus)

// Number to String
let number = 456;
let str1 = String(number);     // "456"
let str2 = number.toString();  // "456"
let str3 = number + "";         // "456" (concatenation with string)

// Boolean Conversion
let bool1 = Boolean(1);         // true
let bool2 = Boolean(0);         // false
let bool3 = Boolean("Hello");   // true
let bool4 = Boolean("");        // false
let bool5 = !!"text";          // true (double NOT operator)

// Falsy values: false, 0, "", null, undefined, NaN
// Truthy values: Everything else

Implicit Type Coercion

// String concatenation with +
console.log("5" + 3);        // "53" (number coerced to string)
console.log("Total: " + 100); // "Total: 100"

// Numeric operations with - * /
console.log("10" - 5);        // 5 (string coerced to number)
console.log("10" * "2");       // 20
console.log("10" / "2");       // 5

// Loose equality (==) performs type coercion
console.log(5 == "5");        // true (string "5" coerced to number 5)
console.log(true == 1);       // true (boolean true coerced to number 1)
console.log(false == 0);      // true (boolean false coerced to number 0)
console.log(null == undefined); // true (special case)

// Strict equality (===) does NOT perform type coercion
console.log(5 === "5");       // false (different types)
console.log(true === 1);      // false
console.log(null === undefined); // false

Avoid Implicit Coercion

Implicit type coercion can lead to unexpected bugs. Always use === and !== for comparisons, and explicitly convert types when needed.

// Unexpected results with implicit coercion
console.log([] == ![]);        // true (confusing!)
console.log(null > 0);         // false
console.log(null == 0);        // false
console.log(null >= 0);        // true (even more confusing!)
